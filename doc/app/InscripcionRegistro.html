def promedio
  promedio_decimal = ("#{examen_medio}".to_f + "#{examen_final}".to_f) / 2
  if promedio_decimal.modulo(2) == 1.5
    promedio_decimal.floor
  else
    promedio_decimal.ceil
  end
end
